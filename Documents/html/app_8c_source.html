71void trafficLightOperation(void)
72{
73 /***********************************************<Normal mode states>*****************************************/
74 ledOn(PEDESTRIANS_RED_LED);//turn pedestrians red led on
75 /*check if the current state is NORMAL_GREEN*/
76 if (system_CurrentState == NORMAL_GREEN)
77 {
78 //execute cars green state
79 carsGreenState();
80 //after the state finished execution
81 //check if i still in the same state
82 //as the state might be changed due to a button pressed
83 // if i still in the same state then change the current state to the next one which is in this example NORMAL_YELLOW_AFTER_GREEN
84 //if the current state changed then do nothing just go to the state which assigned by the button
85 if (system_CurrentState == NORMAL_GREEN)
86 {
87 system_CurrentState = NORMAL_YELLOW_AFTER_GREEN;
88 }
89 }
90 /*normal yellow after green state*/
91 if (system_CurrentState == NORMAL_YELLOW_AFTER_GREEN)
92 {
93 carsYellowState();
94 if (system_CurrentState == NORMAL_YELLOW_AFTER_GREEN)
95 {
96 system_CurrentState = NORMAL_RED;
97 }
98 }
99 /*normal red state */
100 if (system_CurrentState == NORMAL_RED)
101 {
102 carsRedState();
103 if (system_CurrentState == NORMAL_RED)
104 {
105 system_CurrentState = NORMAL_YELLOW_AFTER_RED;
106 }
107 }
108 /*normal yellow after red state*/
109 if (system_CurrentState == NORMAL_YELLOW_AFTER_RED)
110 {
111 carsYellowState();
112 if (system_CurrentState == NORMAL_YELLOW_AFTER_RED)
113 {
114 system_CurrentState = NORMAL_GREEN;
115 }
116 }
117 /***********************************************<Pedestrians mode states>*****************************************/
118 //if the button pressed at green state or one of yellow state
119 if(system_CurrentState == BUTTON_PRESSED_AT_GREEN || system_CurrentState == BUTTON_PRESSED_AT_YELLOW )
120 {
121 ledOn(PEDESTRIANS_RED_LED);
122
123 pedestriansYellowState();
124
125 pedestriansGreenState();
126
127 pedestriansYellowState();
128
129 pedestriansRedState();

135 {
136 pedestriansGreenState();
137
138 pedestriansYellowState();
139
140 pedestriansRedState();
141 system_CurrentState = NORMAL_GREEN;
142 }
143}
