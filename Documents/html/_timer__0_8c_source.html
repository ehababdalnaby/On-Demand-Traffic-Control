127void Timer0_delay_ms(uint32_t delay_ms)
128{
129 //reset Timer 0
130 Timer0_reset();
131 //convert delay time from mile seconds to seconds
132 float64_t neededTimeInsecond = (float64_t)delay_ms/1000;
133 //calculate number of over flows needed to reach the desired time
134 uint32_t numberOfoverFlows = ceil(neededTimeInsecond/Timer0_globalOverFlowTime);
135 //calculate the initial value for #TCNT0 register
136 TCNT0 = TIMER0_NUM_OF_TICKS -(neededTimeInsecond/Timer0_globalTickTime)/numberOfoverFlows;
137 //enable Timer 0 over flow interrupt
138 Timer0_interruptEnable(TIMER0_OVER_FLOW_INT);
139 //start Timer 0 to count
140 Timer0_start();
141 //wait until reaching needed number over flows
142 while(Timer0_globalNumOfOverFlows < numberOfoverFlows);
143 //stop Timer 0 after reaching the desired time.
144 Timer0_stop();
145}
